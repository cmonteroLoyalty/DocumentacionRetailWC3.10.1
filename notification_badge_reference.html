Notification badge

  • Selector name

  • Required imports

  • Extension slots

  • API

  • Implementation example

  • Preferences

Selector name

This widget has the following selector name: <bb-notifications-badge-widget>.

Required imports

The following imports are required to set up and customize this widget:

  • Angular Common module import:

    import { CommonModule } from '@angular/common';

  • Backbase Core module import:

    import { BackbaseCoreModule } from '@backbase/foundation-ang/core';

  • Widget module import:

    import { NotificationsBadgeWidgetModule } from '@backbase/notifications-badge-widget-ang';

Extension slots

This widget has the following extension slots:

  1. bbNotificationsBadgeCustomizable

  2. bbNotificationsListCustomizable

  3. bbNotificationsListItemCustomizable

  4. bbNotificationDetailsCustomizable

  5. bbNotificationDetailsItemCustomizable

  6. bbNotificationsCapabilityIconCustomizable

  7. bbHumanizedDateCustomizable

bbNotificationsBadgeCustomizable

This extension slot includes the following default template:

<ng-template
  bbNotificationsBadgeCustomizable
  let-hostRef
  let-unreadCount="context.unreadCount"
  let-listTransform="context.listTransform"
>
  <div class="bb-notifications-badge-widget">
    <div #popover="ngbDropdown" ngbDropdown placement="bottom-right" (openChange)="hostRef.onOpenChange($event)">
      <button
        bbButton
        [attr.aria-labelledby]="'notifications-badge-label' + hostRef.id"
        color="transparent"
        class="bb-notifications-badge-widget__dropdown-button"
        ngbDropdownToggle
      >
        <bb-icon-ui name="notifications" color="secondary" data-role="notifications-icon" size="lg"></bb-icon-ui>
        <bb-badge-counter-ui
          class="bb-notifications-badge-widget__dropdown-button-counter"
          [count]="unreadCount"
          maxCount="99"
          color="danger"
          data-role="notifications-unread-count"
        ></bb-badge-counter-ui>
      </button>
      <div
        ngbDropdownMenu
        class="bb-notifications-badge-widget__dropdown-menu dropdown-menu-unstyled"
        data-role="notifications-list-wrapper"
      >
        <div
          class="bb-notifications-badge-widget__dropdown-menu-list"
          data-role="notifications-dropdown-menu-list"
          [style.transform]="listTransform"
        >
          <div class="card bb-notifications-badge-widget__list-wrapper">
            <bb-notifications-list
              *ngIf="popover.isOpen()"
              (navigation)="popover.close()"
              [preselectedNotification]="hostRef.preselectedNotification$ | async"
            ></bb-notifications-list>
          </div>
        </div>
      </div>
      <span
        i18n="Notifications badge aria label@notifications.badge.aria-label"
        [attr.id]="'notifications-badge-label' + hostRef.id"
        aria-live="polite"
        aria-relevant="text"
        class="sr-only"
      >
        { unreadCount || 0, plural, =0 { no new notifications } =1 { {{ unreadCount }} new notification} other {
        {{ unreadCount }} new notifications } }
      </span>
    </div>
  </div>
</ng-template>

<ng-template
  bbNotificationsListCustomizable
  let-hostRef
  let-notifications="context.notifications"
  let-error="context.error"
  let-isLoading="context.isLoading"
  let-totalCount="context.totalCount"
  let-pageSize="context.pageSize"
  let-newNotificationsAvailable="context.newNotificationsAvailable"
  let-unreadCount="context.unreadCount"
  let-markAllAsReadError="context.markAllAsReadError"
  let-pageFrom="context.pageFrom"
>
  <ng-container *ngIf="!hostRef.selectedNotification">
    <header data-role="notifications-list-header" class="card-header bb-stack">
      <span class="bb-stack__item">
        <span class="bb-heading-3" i18n="Notifications modal label@@notifications.modalLabel">Notifications</span>
      </span>

        bbButton
        class="bb-stack__item bb-stack__item--push-right"
        data-role="settings-button"
        color="link"
        circle="true"

        i18n-aria-label="Aria-label for settings button@@notifications.notification-settings.aria-label"
        (click)="hostRef.onNotificationsSettingsOpen()"
      >
        <bb-icon-ui class="text-dark" name="settings" size="md"></bb-icon-ui>
      </button>
    </header>
    <button
      bbButton
      *ngIf="newNotificationsAvailable"
      role="status"
      color="primary"
      [block]="true"
      i18n="Label for new notifications loading button@@notifications.new-notifications-available.button"
      aria-label="New Notification available. Click to refresh"
      i18n-aria-label="
        Aria-label for new notifications loading button@@notifications.new-notifications-available.aria-label"
      class="bb-notifications-badge-widget__load-new-button"
      data-role="notifications-load-new-btn"
      (click)="hostRef.onLoadNewNotifications()"
    >
      <span>New Notification available. Click to refresh</span>
      <bb-icon-ui name="loading" size="md" cropped="true"></bb-icon-ui>
    </button>
    <bb-alert-ui
      *ngIf="markAllAsReadError"
      class="bb-notifications-badge-widget__alert"
      data-role="mark-all-read-error-alert"
      title="Error! Couldn’t mark all as read."
      i18n-title="Label for mark all as read error@@notifications.mark-all-read.alert.title"
      aria-label="Error! Couldn’t mark all as read."
      i18n-aria-label="Aria-label for mark all as read error@@notifications.mark-all-read.alert.aria-label"
      modifier="error"
      [dismissible]="false"
    >
    </bb-alert-ui>
    <div class="bb-notifications-badge-widget__all-read-button" *ngIf="unreadCount > 0">
      <button
        bbButton
        data-role="mark-all-read-button"
        class="bb-subtitle"
        color="link-text"
        aria-label="Mark all notifcations as read"
        i18n-aria-label="Aria-label for mark all as read button@@notifications.mark-all-read.aria-label"
        (click)="hostRef.onReadAllNotifications()"
      >
        <bb-icon-ui name="playlist-add-check" size="md" cropped="true"></bb-icon-ui>
        <span i18n="Label for mark all as read button@@notifications.mark-all-read.label">Mark all as read</span>
      </button>
    </div>
  </ng-container>

  <ng-container *ngIf="notifications; else loadingState">
    <ng-container *ngIf="!error && notifications.length">
      <div data-role="notifications-list-inner-wrapper" [ngClass]="hostRef.selectedNotification ? 'd-none' : null">
        <p
          aria-live="polite"
          class="sr-only"
          i18n="Label for new loaded notifications available@@notifications.new-loaded-notifications-available"
        >
          { notifications.length, plural, =1 {new notification of {{ totalCount }}available} other {
          {{ notifications.length }} new notifications of {{ totalCount }}available } }
        </p>
        <ul
          tabindex="-1"
          class="bb-notifications-badge-widget__list bb-list bb-list--flush-lg"
          data-role="notifications-list-body"
        >
          <li class="bb-list__item" *ngFor="let notification of notifications; trackBy: hostRef.trackById">
            <bb-notifications-list-item
              [notification]="notification"
              class="bb-nt-{{ notification.id }}"
              (click)="hostRef.selectNotification($event, notification)"
              (openDetails)="hostRef.selectNotification($event, notification, true)"
            >
            </bb-notifications-list-item>
          </li>
        </ul>

        <footer class="card-footer">
          <ng-container *ngIf="totalCount > pageSize">
            <div
              *ngIf="notifications.length < totalCount; else LoadMoreDisabled"
              class="bb-button-bar bb-button-bar--center"
            >
              <bb-load-button-ui
                i18n="Load more button@@notifications.notification-list.load-more"
                class="bb-button-bar__button"
                color="link"
                [size]="pageSize"
                [from]="pageFrom"
                [totalCount]="totalCount"
                [isLoading]="isLoading"
                [block]="false"
                (paramChange)="hostRef.onLoadMore(totalCount, $event)"
              >
                Load More
              </bb-load-button-ui>
            </div>
            <ng-template #LoadMoreDisabled>
              <div
                data-role="notification-list-load-more-disabled-text"
                class="bb-subtitle bb-text-support bb-text-align-center"
                i18n="Load more disabled text@@notifications.notification-list.load-more-disabled"
              >
                You’ve reached the end of the list.
              </div>
            </ng-template>
          </ng-container>
        </footer>
      </div>

      <bb-notification-details
        *ngIf="hostRef.selectedNotification"
        [notification]="hostRef.selectedNotification"
        (navigateBack)="hostRef.unselectNotification()"
        (delete)="hostRef.deleteNotification()"
      ></bb-notification-details>
    </ng-container>
  </ng-container>

  <div role="status">
    <div class="card-body" *ngIf="!notifications?.length || error">
      <bb-empty-state-ui
        *ngIf="error"
        data-role="notifications-badge-error-state"
        class="bb-state-container"
        iconSize="xxl"
        iconModifier="error-outline"
        title="Notifications couldn’t load"
        i18n-title="Notifications are not loading title@@notifications.error-state.title"
        subtitle="Try to reload the page or contact the system administrator."
        i18n-subtitle="Notifications are not loading subtitle@@notifications.error-state.subtitle"
      >
      </bb-empty-state-ui>

      <bb-empty-state-ui
        *ngIf="notifications && !notifications.length && !error"
        data-role="notifications-badge-empty-state"
        class="bb-state-container"
        i18n-title="Notifications empty state message title@@notifications.state.no-notifications.title"
        title="No Notifications"
        iconModifier="notifications-off"
        iconSize="xxl"
        i18n-subtitle="Notifications empty state message@@notifications.state.no-notifications.subtitle"
        subtitle="You don't have any notifications to be displayed"
      ></bb-empty-state-ui>
    </div>
  </div>

  <ng-template #loadingState>
    <bb-loading-indicator-ui
      loaderSize="md"
      text="Loading..."
      i18n-test="Loading title@@notifications.state.loading.title"
    ></bb-loading-indicator-ui>
  </ng-template>
</ng-template>

          i18n="Notifications time less than an hour ago date@@notifications.humanized-date.less-than-an-hour-ago"
        >
          { hostRef.minutesFromNow, plural, =0 {now} other {{{hostRef.minutesFromNow}} min ago} }
        </ng-container>

        <ng-template #moreThanHourAgo>
          {{ date | bbDate: 'shortTime' }}
        </ng-template>
      </ng-container>

      <ng-template #today i18n="Notifications today date@@notifications.humanized-date.today"> today </ng-template>
    </time>

    <time
      *ngIf="hostRef.isYesterday"
      class="bb-subheader bb-subheader--regular"
      i18n="Notifications yesterday date@@notifications.humanized-date.yesterday"
    >
      yesterday
    </time>

    <time
      *ngIf="!hostRef.isToday && !hostRef.isYesterday"
      class="bb-subheader bb-subheader--regular"
      [attr.aria-label]="date | bbDate: 'mediumDate'"
      role="text"
    >
      {{ date | bbDate: 'MM/dd/yy' }}
    </time>
  </ng-container>
</ng-template>

Display relative time for today

Whether the exact date of receiving a notification should be changed to a more convenient relative format. If this preference is set to true, then the following rules apply:

  • If the receipt time is less than 1 minute ago, the timestamp changes to Now.

  • If the receipt time is more than 1 minute but less than 1 hour ago, then the number of minutes is displayed. For example, 59 min ago.

  • If the receipt time is more than 1 hour but less than 1 day ago, then the HH:MM AM/PM format is used. For example, 10:00 AM.

  • If the receipt time is 1 day ago, the timestamp changes to Yesterday.

  • If the receipt time is more than 1 day ago, then the DD/MM/YYYY format is used. For example, 10/03/2020.
