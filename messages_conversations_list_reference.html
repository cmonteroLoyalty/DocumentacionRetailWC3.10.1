Selector name

This widget has the following selector name: bb-messages-conversations-list-widget-ang.

Required imports

The following imports are required to set up and customize this widget:

  • Angular Common module import:

    import { CommonModule } from '@angular/common';

  • Backbase Core module import:

    import { BackbaseCoreModule } from '@backbase/foundation-ang/core';

  • Widget module import:

    import { MessagesConversationsListWidgetModule } from '@backbase/messages-conversations-list-widget-ang';

Extension slots

This widget has the following extension slots:

  1. bbMessagesConversationsListCustomizable

  2. bbConversationsListTableCustomizable

  3. bbConversationsListTableRowCustomizable

  4. bbConversationsListTableHeadCustomizable

bbMessagesConversationsListCustomizable

This extension slot includes the following default template:

<ng-template
  bbMessagesConversationsListCustomizable
  let-host
  let-conversations="context.conversations"
  let-mailboxType="context.mailboxType"
  let-itemsPerPage="context.itemsPerPage"
>
  <div class="bb-messages-conversations-list__container card card-lg">
    <ng-container *ngIf="conversations?.data; else loadingState">
      <ng-container *ngIf="!host.errorResponse; else errorState">
        <ng-container *ngIf="conversations.data.length; else emptyState">
          <bb-conversations-list-table-ui
            [conversations]="conversations.data"
            [reset]="host.resetSelection"
            [mailboxType]="host.mailboxType | async"
            [customerServiceTitle$]="host.customerServiceTitle"
            (remove)="host.removeConversations($event)"
            (read)="host.markAsReadUnread($event)"
            (select)="
              mailboxType === host.mailboxTypeEnum.drafts ? host.openDraft($event) : host.openConversation($event)
            "
            (update)="host.updateConversations()"
          ></bb-conversations-list-table-ui>
          <div class="p-3 text-center">
            <bb-load-button-ui
              [from]="host.from.value.from"
              [isLoading]="host.isLoading"
              [size]="itemsPerPage"
              [totalCount]="conversations.totalCount"
              (paramChange)="host.setLoadingParams($event)"
              data-role="load-more-button"
              i18n="Load more button name@@messages.messagesConversationsList.loadMoreButton"
              >Load more</bb-load-button-ui
            >
          </div>
        </ng-container>
      </ng-container>
    </ng-container>
  </div>

  <ng-template #emptyState>
    <div class="bb-state-container">
      <ng-container [ngSwitch]="mailboxType">
        <ng-container *ngSwitchCase="host.mailboxTypeEnum.drafts">
          <bb-empty-state-ui
            i18n-title="Empty state title@@messages.messagesConversationsList.draftsEmptyStateTitle"
            title="No Drafts"
            i18n-subtitle="Empty state subtitle@@messages.messagesConversationsList.draftsEmptyStateSubTitle"
            subtitle="You don't have any drafts to be displayed."
            iconModifier="email"
            iconColor="muted"
          ></bb-empty-state-ui>
        </ng-container>
        <ng-container *ngSwitchCase="host.mailboxTypeEnum.outbox">
          <bb-empty-state-ui
            i18n-title="Empty state title@@messages.messagesConversationsList.emptyStateTitle"
            title="No Sent Messages"
            i18n-subtitle="Empty state subtitle@@messages.messagesConversationsList.emptyStateSubtitle"
            subtitle="You don't have any sent messages to be displayed."
            iconModifier="email"
            iconColor="muted"
          ></bb-empty-state-ui>
        </ng-container>
        <ng-container *ngSwitchDefault>
          <bb-empty-state-ui
            i18n-title="Empty state title@@messages.messagesConversationsList.simpleEmptyStateTitle"
            title="No Conversations"
            i18n-subtitle="Empty state subtitle@@messages.messagesConversationsList.simpleEmptyStateSubTitle"
            subtitle="You don't have any conversations to be displayed."
            iconModifier="email"
            iconColor="muted"
          ></bb-empty-state-ui>
        </ng-container>
      </ng-container>
    </div>
  </ng-template>

  <ng-template #errorState>
    <bb-messages-request-error [error]="host.errorResponse"></bb-messages-request-error>
  </ng-template>

  <ng-template #loadingState>
    <bb-loading-indicator-ui class="py-4" loaderSize="md"></bb-loading-indicator-ui>
  </ng-template>
</ng-template>
