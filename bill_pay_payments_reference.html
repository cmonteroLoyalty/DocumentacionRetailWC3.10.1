Selector name

This widget has the following selector name: <bb-billpay-payments-widget>.

Required imports

The following imports are required to set up and customize this widget:

  • Widget module import

    import { BillpayPaymentsWidgetModule } from '@backbase/billpay-payments-widget-ang';
    NOTE

    All Bill Pay widgets are closely related to each other, so it is recommended to use them together.

bbBillpayPaymentsPageCustomizable

Configure this extension slot using:

<ng-container bbBillpayPaymentsPageCustomizable
  [bbHostRef]="this"
  [bbTemplateContext]="payments$ | async"
>
</ng-container>

<ng-template bbBillpayPaymentsPageCustomizable
  let-hostRef
  let-payments="context"
>
  <ng-container *ngIf="payments; else loadingState">
    <ng-container *ngIf="!hostRef.hasLoadingError; else errorState">
      <ng-container *ngIf="payments.length; else emptyState">
        <section class="{{ hostRef.isThemeV2Enabled ? '' : 'container-medium'}}">
          <bb-billpay-payments-list
            [payments]="payments"
          ></bb-billpay-payments-list>
        </section>
      </ng-container>
    </ng-container>
  </ng-container>

  <ng-template #loadingState>
    <ng-container *ngIf="!hostRef.isThemeV2Enabled; else themeV2LoadingState">
      <bb-loading-indicator-ui loaderSize="lg">
      </bb-loading-indicator-ui>
    </ng-container>
    <ng-template #themeV2LoadingState>
      <div class="card card-lg">
        <div class="card-body">
          <bb-loading-indicator-ui loaderSize="lg">
          </bb-loading-indicator-ui>
        </div>
      </div>
    </ng-template>
  </ng-template>

  <ng-template #errorState>

        title="Payments list is not loading"
        i18n-title="Payments list is not loading title@@billpay-payments.error-state.title"
        subtitle="Try to reload the page or contact support"
        i18n-subtitle="Payments list is not loading subtitle@@billpay-payments.error-state.subtitle">
      </bb-empty-state-ui>
    </ng-container>
    <ng-template #themeV2ErrorState>
      <div class="card card-lg">

            <bb-empty-state-ui
              data-role="error-state"

bbBillpayPaymentsGroupCustomizable

Configure this extension slot using:

<ng-container bbBillpayPaymentsGroupCustomizable
  [bbHostRef]="this"
  [bbTemplateContext]="paymentsGroup"
>
</ng-container>

<ng-template bbBillpayPaymentsGroupCustomizable
  let-hostRef
  let-paymentsGroup="context"
>

  <ng-container *ngIf="!hostRef.isThemeV2Enabled; else themeV2PaymentGroup">
    <div
      class="text-support mb-2 pb-1"
      data-role="date-label"
    >
      <ng-container
        *ngIf="hostRef.getDateLabel(paymentsGroup?.date) as date; else formattedDate"
        i18n="Date label for payments group@@billpay-payments.payments-group.date-label">
        { date, select, yesterday {Yesterday} today {Today} tomorrow {Tomorrow} }
      </ng-container>
      <ng-template #formattedDate>{{ paymentsGroup?.date | bbDate: "MMMM d" }}</ng-template>
    </div>

    <div class="card shadow-sm">
      <ul class="list-group list-group-flush">
        <li
          data-role="list-group-item"
          *ngFor="let payment of paymentsGroup?.payments"
          class="list-group-item"
          role="button"
        >
          <bb-billpay-payment-card
            [payment]="payment"
            isNegativeAmount="true"
            (click)="hostRef.navigateToPayment(payment.id, payment.recurring)"
          ></bb-billpay-payment-card>
        </li>
      </ul>
    </div>
  </ng-container>
  <ng-template #themeV2PaymentGroup>
    <section>
      <div
        role="heading"

          i18n="Date label for payments group@@billpay-payments.payments-group.date-label">
          { date, select, yesterday {Yesterday} today {Today} tomorrow {Tomorrow} }
        </ng-container>

      </div>
      <ul class="bb-list">

          data-role="list-group-item"
          class="bb-list__item bb-list__item--no-separator"
          *ngFor="let payment of paymentsGroup?.payments"
          role="button"

          <bb-billpay-payment-card
            [payment]="payment"
            isNegativeAmount="true"
            (click)="hostRef.navigateToPayment(payment.id, payment.recurring)"
          ></bb-billpay-payment-card>

      </ul>

  </ng-template>
</ng-template>
