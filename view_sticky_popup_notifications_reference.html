View sticky and pop-up notifications

This page includes the following information:

  • Selector name

  • Required imports

  • Extension slots

  • API

  • Implementation example

  • Preferences

Selector name

This widget has the following selector name: <bb-notifications-popups-widget>.

Required imports

The following imports are required to set up and customize this widget:

  • Angular Common module import:

    import { CommonModule } from '@angular/common';

  • Backbase Core module import:

    import { BackbaseCoreModule } from '@backbase/foundation-ang/core';

  • Widget module import:

    import { NotificationsPopupsWidgetModule } from '@backbase/notifications-popups-widget-ang';

Extension slots

This widget has the following extension slots:

  1. bbNotificationsStickyListItemCustomizable

  2. bbNotificationsStickyListCustomizable

  3. bbNotificationsPopupsListCustomizable

  4. bbNotificationsPopupsItemHeaderCustomizable

  5. bbNotificationsPopupsItemMessageCustomizable

bbNotificationsStickyListItemCustomizable

This extension slot includes the following default template:

<ng-template bbNotificationsStickyListItemCustomizable let-host let-notification="context">
  <bb-alert-ui
    [dismissible]="true"
    [modifier]="host.parseNotificationSeverityLevel(notification?.level)"
    [title]="notification?.title || ''"
    (close)="host.onClose(notification)"
  >
    <div
      class="bb-subtitle"
      data-role="notification-sticky-message"
      [innerHTML]="notification?.message | interpolateLink: notification:true:'alert-link'"
    ></div>
  </bb-alert-ui>
</ng-template>

bbNotificationsStickyListCustomizable

This extension slot includes the following default template:

<ng-template bbNotificationsStickyListCustomizable let-host let-notifications="context.notifications">
 <bb-notifications-sticky-list-item
   (focusin)="host.onFocus($event)"
   *ngFor="let item of notifications; trackBy: host.trackById"
   [notification]="item"
   (close)="host.onReadNotification($event)"
 ></bb-notifications-sticky-list-item>

bbNotificationsPopupsListCustomizable

This extension slot includes the following default template:

<ng-template bbNotificationsPopupsListCustomizable let-hostRef let-notifications="context">
  <bb-notifications-popups-item
    *ngFor="let notification of notifications; index as i"
    [notification]="notification"
    [notificationTimeout]="hostRef.getNotificationTimeoutByIndex(i, notifications.length)"
  >
  </bb-notifications-popups-item>
</ng-template>

bbNotificationsPopupsItemHeaderCustomizable

This extension slot includes the following default template:

<ng-template bbNotificationsPopupsItemHeaderCustomizable let-hostRef let-notification="context">
  <span class="sr-only"> {{ notification?.level || '' }}, </span>
  <span> {{ notification?.title || '' }} </span>
</ng-template>

bbNotificationsPopupsItemMessageCustomizable

This extension slot includes the following default templates:

<ng-template bbNotificationsPopupsItemMessageCustomizable let-hostRef let-notification="context">
  <ng-container
    *ngTemplateOutlet="(hostRef.popupMessageLineAmount$ | async) ? truncatedMessageContent : messageContent"
  ></ng-container>
  <bb-notifications-capability-icon [origin]="notification?.origin"></bb-notifications-capability-icon>
  <button
    type="button"
    *ngIf="hostRef.isMessageTruncated && (hostRef.displayShowMoreButtonInPopup$ | async)"
    class="btn-link-text bb-text-bold bb-notification__details-button"
    (click)="hostRef.openNotificationDetails($event)"
    aria-label="Show more notification details"
    i18n="Notification pop-up show more button@notification.popup.show-more.button"
    i18n-aria-label="Aria-label for show more button@@notification.popup.show-more.button.aria-label"
  >
    Show more
  </button>

  <ng-template #messageContent>
    <div class="bb-block bb-block--sm">
      <div
        bbIsElementContentTruncated
        [hostRef]="hostRef"
        [innerHTML]="notification?.message | interpolateLink: notification"
      ></div>
    </div>
  </ng-template>

  <ng-template #truncatedMessageContent>
    <bb-ellipsis-ui [tooltipTriggers]="''" [lineClamp]="hostRef.popupMessageLineAmount$ | async">
      <ng-container *ngTemplateOutlet="messageContent"></ng-container>
    </bb-ellipsis-ui>
  </ng-template>
</ng-template>

API

The following table shows the public class members of the host component that you access from each extension slot:

Component Extension slot identifier Public class member Description

NotificationsPopupsItemComponent

bbNotificationsPopupsItemHeaderCustomizable, bbNotificationsPopupsItemMessageCustomizable

@Input() notification: NotificationStreamItem

The notification popup item.

@Input() notificationTimeout: number

The delay timeout for closing a notification.

@ViewChild popupHeaderTemplate: TemplateRef<any>

The notification header template.

@ViewChild popupMessageTemplate: TemplateRef<any>

The notification message template.

readonly allowedRoutes$: Observable<string[]>

Routes available for navigation.

readonly autofocus$

The stream of the widget preference with a flag that defines if the notification popup should be auto-focused.

readonly displayShowMoreButtonInPopup$

The stream of the widget property with a flag that defines if the Show More button should be displayed.

readonly popupMessageLineAmount$

The stream of the widget preference with the number of lines to show as a notification message.

isMessageTruncated

The boolean flag that shows if the notification message text exceeds the number of visible lines and is truncated.

openNotificationDetails(event: Event): void

The method to open notification details.

setIsContentTruncated(isMessageTruncated: boolean): void

Sets the truncation status of the notification message.

NotificationsPopupsListComponent

bbNotificationsPopupsListCustomizable

readonly popups$: Observable<NotificationStreamItem[]>

The notification popups stream data in the reversed order.

getNotificationTimeoutByIndex(index: number, length: number): number

The method to get the close notification timeout value for each popup.

NotificationsStickyListComponent

bbNotificationsStickyListCustomizable

readonly notifications$: Observable<NotificationsStream>

The sticky notification stream data.

onReadNotification(notification: NotificationStreamItem | string): void

The method to send the mark-notification-as-read request by ID.

trackById(_index: number, notification: NotificationItem): string

The method to differentiate notifications by IDs.

onFocus(event: { relatedTarget: HTMLElement }): void

The method to change the focus.

NotificationsStickyListItemComponent

bbNotificationsStickyListItemCustomizable

@Input() notification: NotificationStreamItem

The notification stream item.

@Output() close: EventEmitter<NotificationStreamItem | string>

The notification stream item close event that emits the notification stream item ID.

onClose(notification: NotificationStreamItem | string): void

The method to handle the close event from the alert component.

parseNotificationSeverityLevel(level: SeverityLevel): NotificationSeverityType

The method to parse the severity level into the alert modifier.

Implementation example

The following examples show the implementation of this widget:

  • Wrapper widget module

    notifications-popups-extended.module.ts
    import { NgModule } from '@angular/core';
import { CommonModule } from '@angular/common';
import { BackbaseCoreModule } from '@backbase/foundation-ang/core';
import { NotificationsPopupsWidgetModule } from '@backbase/notifications-popups-widget-ang';
import { NotificationsPopupsExtendedComponent } from './notifications-popups-extended.component';

@NgModule({
  imports: [
    CommonModule,
    BackbaseCoreModule.withConfig({
      classMap: { NotificationsPopupsExtendedComponent },
    }),
    NotificationsPopupsWidgetModule,
  ],
  declarations: [NotificationsPopupsExtendedComponent],
  exports: [NotificationsPopupsExtendedComponent],
})
export class NotificationsPopupsExtendedModule {}

  • Wrapper widget component

    notifications-popups-extended.component.ts
    import { Component, EventEmitter, Output } from '@angular/core';
import { NotificationRouting } from '@backbase/notifications-common-ang';

@Component({
  selector: 'bb-notifications-popups-extended',
  templateUrl: './notifications-popups-extended.component.html',
})
export class NotificationsPopupsExtendedComponent {
  @Output() notificationRouting = new EventEmitter<NotificationRouting>();

  onNotificationRoutingChange(rounting: NotificationRouting) {
    this.notificationRouting.emit(rounting);
  }
}

  • Custom template

    notifications-popups-extended.component.html
    <bb-notifications-popups-widget (notificationRouting)="onNotificationRoutingChange($event)">
  <ng-template bbNotificationsPopupsListCustomizable let-hostRef let-notifications="context">
    <!-- place your customization here -->
  </ng-template>

  <ng-template bbNotificationsStickyListCustomizable let-hostRef let-notifications="context.notifications">
    <!-- place your customization here -->
  </ng-template>
</bb-notifications-popups-widget>
