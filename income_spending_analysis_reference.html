View income and spending by category

This page includes the following information:

  • Selector name

  • Required imports

  • Extension slots

  • Implementation example

  • Preferences

Selector name

This widget has the following selector name: <bb-income-spending-analysis-widget>.

Required imports

The following imports are required to set up and customize this widget:

  • Widget module import

    Import the following modules into the module of your wrapper widget:

    import { IncomeSpendingAnalysisWidgetModule } from '@backbase/retail-ang/pfm';

  • Widget component import

    Import the following components into the module of your wrapper widget:

    import { IncomeSpendingAnalysisWidgetComponent } from '@backbase/retail-ang/pfm';

  • Data module imports

    Import the following data modules into the module of your wrapper widget:

    import { TransactionsDataHttpModule } from '@backbase/data-ang/transactions';

Extension slots

This widget has the following extension slots:

  1. bbIncomeSpendingAnalysisCustomizable

  2. bbIncomeSpendingAnalysisPeriodSelectorCustomizable

  3. bbIncomeSpendingAnalysisMonthSelectorCustomizable

  4. bbIncomeSpendingAnalysisChartWrapperCustomizable

  5. bbIncomeSpendingAnalysisChartCustomizable

  6. bbIncomeSpendingAnalysisBarChartWrapperCustomizable

  7. bbIncomeSpendingAnalysisCategoriesListCustomizable

bbIncomeSpendingAnalysisCustomizable

This extension slot includes the following default template:

<ng-template
  bbIncomeSpendingAnalysisCustomizable
  let-hostRef
  let-analysisData="context.analysisData"
  let-analysisPeriodData="context.analysisPeriodData"
  let-colors="context.colors"
  let-isSpending="context.isSpending"
  let-chartOptions="context.chartOptions"
>
  <ng-container *ngIf="!hostRef.error; else errorState">
    <div class="card">
      <div class="card-body">
        <ng-container
          bbIncomeSpendingAnalysisMonthSelectorCustomizable
          [bbHostRef]="hostRef"
        ></ng-container>

        <ng-container
          *ngIf="hostRef.selectedTab === 1"
          bbIncomeSpendingAnalysisPeriodSelectorCustomizable
          [bbHostRef]="hostRef"
        ></ng-container>
        <ng-container *ngIf="(arrangementIds | async)?.length; else emptyState">
          <ng-container *ngIf="hostRef.selectedTab === 1">
            <ng-container *ngIf="analysisData | async as data ; else loadingState">
              <ng-container *ngIf="data.items && data.items.length; else emptyState">
                <ng-container
                  bbIncomeSpendingAnalysisChartWrapperCustomizable
                  [bbHostRef]="hostRef"
                  [bbTemplateContext]="{ data: data, colors: colors, isSpending: isSpending }"
                ></ng-container>
              </ng-container>
            </ng-container>
          </ng-container>
          <ng-container *ngIf="hostRef.selectedTab !== 1">
            <ng-container *ngIf="analysisPeriodData | async as data; else loadingState">
              <ng-container *ngIf="!hostRef.isLoading else loadingState">
                <ng-container *ngIf="data.series && data.series.length; else emptyState">
                  <ng-container
                    bbIncomeSpendingAnalysisBarChartWrapperCustomizable
                    [bbHostRef]="hostRef"
                    [bbTemplateContext]="{ data: data, chartOptions: chartOptions }"
                  ></ng-container>
                </ng-container>

                <ng-container *ngIf="data.categoryData.items.length else emptyState">
                  <ng-container
                    bbIncomeSpendingAnalysisCategoriesListCustomizable
                    [bbHostRef]="hostRef"
                    [bbTemplateContext]="{ data: data.categoryData, isSpending: isSpending }"
                  ></ng-container>
                </ng-container>
              </ng-container>
            </ng-container>
          </ng-container>
        </ng-container>

      </div>
    </div>
  </ng-container>

  <ng-template bbIncomeSpendingAnalysisMonthSelectorCustomizable let-hostRef>
    <bb-month-selector-ui
      [selectedTab]="hostRef.selectedTab"
      (change)="hostRef.onPeriodRangeChange($event)"
    ></bb-month-selector-ui>
  </ng-template>

  <ng-template bbIncomeSpendingAnalysisPeriodSelectorCustomizable let-hostRef>
    <div class="bb-stack bb-stack--center bb-block bb-block--lg">
      <bb-period-selector-ui
        [periodStart]="hostRef.startDate"
        [periodEnd]="hostRef.endDate"
        [periodFormatters]="hostRef.formatters"
        (periodChange)="hostRef.onPeriodChange($event)"
      ></bb-period-selector-ui>
    </div>
  </ng-template>

  <ng-template
    bbIncomeSpendingAnalysisChartWrapperCustomizable
    let-hostRef
    let-data="context.data"
    let-colors="context.colors"
    let-isSpending="context.isSpending"
  >
    <div class="bb-stack bb-stack--center position-relative chart-donut chart-donut-large chart-donut-container-inflated">
      <div class="absolute-center">
        <div class="text-center text-muted font-weight-bold chart-donut-center-word">
          <span
            *ngIf="isSpending"
            i18n="Total spending|Label for total spending amount@@incomeSpendingAnalysis.label.total.spending"
          >
            Total spending
          </span>
          <span
            *ngIf="!isSpending"
            i18n="Total income|Label for total income amount@@incomeSpendingAnalysis.label.total.income"
          >
            Total income
          </span>
        </div>
        <div class="h6 text-center">
          <bb-amount-ui
            [amount]="data.total.amount"
            [currency]="data.total.currencyCode"
          ></bb-amount-ui>
        </div>
      </div>
      <ng-container
        bbIncomeSpendingAnalysisChartCustomizable
        [bbHostRef]="hostRef"
        [bbTemplateContext]="{ data: data, colors: colors, isSpending: isSpending }"
      ></ng-container>
    </div>
  </ng-template>

  <ng-template
    bbIncomeSpendingAnalysisChartCustomizable
    let-hostRef
    let-data="context.data"
    let-colors="context.colors"
  >
    <bb-donut-chart-ui
      [data]="data.series"
      [colors]="colors"
      [relevancePercentage]="hostRef.relevance"
      [tooltipTmpl]="donutChartTooltip"
      [labelTmpl]="donutChartLabel"
    ></bb-donut-chart-ui>
  </ng-template>

  <ng-template #errorState>
    <bb-empty-state-ui
      title=""
      [subtitle]="error"
      iconModifier="error"
      iconSize="xxl"
      iconColor="gray"
      iconClasses="text-muted"
    ></bb-empty-state-ui>
  </ng-template>

  <ng-template #loadingState>
    <bb-loading-indicator-ui
      i18n-text="Loading analysis data|Text to display while loading income/spending analysis data@@incomeSpendingAnalysis.state.loadingData"
      text="Loading analysis data...">
    </bb-loading-indicator-ui>
  </ng-template>

  <ng-template #emptyState>
    <div class="bb-stack bb-stack--center">
      <bb-empty-state-ui
      i18n-title="No data title|Title of a message that is shown when there is no data to display@@incomeSpendingAnalysis.state.noResults.title"
      title="No Data Found"
      i18n-subtitle="No data subtitle|Subtitle of a message that is shown when there is no data to display@@incomeSpendingAnalysis.state.noResults.subtitle"
      subtitle="You don't have any data to be displayed. Please try selecting a different period."
      iconSize="xxl"
      iconModifier="donut-small"
      class="bb-text-align-center"
    ></bb-empty-state-ui>
    </div>
  </ng-template>

  <ng-template bbIncomeSpendingAnalysisCategoriesListCustomizable let-hostRef let-data="context.data" let-isSpending="context.isSpending">
    <div class="list-group-item rounded-0 border-left-0 border-right-0 border-top-0 list-item-transaction-accordion pt-3">
      <div class="transaction-item row align-items-center" role="row">
        <div class="col-6">
          <span *ngIf="isSpending" class="h5" i18n="Spending per category|Categories list spending label@@categories.list.spending.label">
            Spending per category
          </span>
          <span *ngIf="!isSpending" class="h5" i18n="Income per category|Categories list income label@@categories.list.income.label">
            Income per category
          </span>
        </div>
        <div class="col-6">
          <div class="transaction-amount text-right h5 m-0">
            <span i18n="Total amount|Total amount label@@categories.list.total.label">
              Total amount
            </span>
            <bb-amount-ui [amount]="data.total.amount"></bb-amount-ui>
          </div>
        </div>
      </div>
    </div>
    <div class="list-group-item rounded-0 border-left-0 border-right-0 list-item-transaction-accordion py-3" *ngFor="let item of data.items">
      <div class="transaction-item row d-flex align-items-center" role="row">
        <div class="transaction-category-icon col-6 d-flex align-items-center">
          <bb-icon-ui
            size="sm"
            backgroundType="circle"
            name="category-{{ hostRef.transformCategoryToIconName(item.category) }}"
          ></bb-icon-ui>
          <strong>
            <span class="bb-transaction-category-name px-3">{{item.category}}</span>
          </strong>
        </div>
        <div class="col-6">
          <div class="transaction-amount text-right">
            <bb-amount-ui [amount]="item.totalAmount.amount"></bb-amount-ui>
          </div>
        </div>
      </div>
    </div>
  </ng-template>

  <ng-template bbIncomeSpendingAnalysisBarChartWrapperCustomizable let-chartOptions="context.chartOptions" let-data="context.data">
    <div class="d-block chart-bar">
      <ngx-charts-bar-vertical
        [scheme]="{domain: chartOptions.colors}"
        [results]="data.series"
        [gradient]="chartOptions.gradient"
        [xAxis]="chartOptions.showXAxis"
        [yAxis]="chartOptions.showYAxis"
        [legend]="chartOptions.showLegend"
        [showXAxisLabel]="chartOptions.showXAxisLabel"
        [showYAxisLabel]="chartOptions.showYAxisLabel"
        [showGridLines]="true"
        [roundEdges]="chartOptions.roundEdges"
      >
        <ng-template #tooltipTemplate let-model="model">
          <div class="my-2">
            <span class="mr-2">{{model.name}}</span>
            <bb-amount-ui class="ml-2" [amount]="model.value"></bb-amount-ui>
          </div>
        </ng-template>
      </ngx-charts-bar-vertical>
    </div>
  </ng-template>

  <ng-template
    #donutChartLabel
    let-data="data"
    let-color="color"
    let-label="label"
    let-icon="icon"
  >
    <svg:g
      [attr.transform]="hostRef.getAttrTransform(data, -hostRef.iconWidth, hostRef.getTransformOffsetY(data))"
      [style.transform]="hostRef.getStyleTransform(data, -hostRef.iconWidth, hostRef.getTransformOffsetY(data))"
      [style.fill]="color"
    >
      <svg:text
        #bbChartLabelIcon
        class="bb-transaction-category bb-transaction-category-icon"
        [ngClass]="hostRef.getCategoryClass(data.data.cssClass)"
      >
        {{ icon }}
      </svg:text>
    </svg:g>
    <svg:g
      [attr.transform]="hostRef.getAttrTransform(data, 0, 0)"
      [style.transform]="hostRef.getStyleTransform(data, 0, 0)"
      [style.transition]="hostRef.getTextTransition()"
    >
      <svg:text
        #bbChartLabelText
        class="font-weight-bold"
        [class.animation]="true"
        dy=".35em"
        [style.fill]="color"
        [style.textAnchor]="'middle'"
        [style.shapeRendering]="'crispEdges'"
      >
        {{ label }}
      </svg:text>
    </svg:g>
  </ng-template>

  <ng-template
    #donutChartTooltip
    let-name="name"
    let-cssClass="cssClass"
    let-totalAmount="totalAmount"
    let-portion="portion"

    let-arrowClass="arrowClass"

    <div
      class="row pb-3 position-relative chart-tooltip chart-tooltip-default text-left"
      [ngClass]="arrowClass"
    >
      <span class="d-flex chart-tooltip-analysis-category text-muted">
        <span
          *ngIf="hostRef.relevance >= portion"
          class="d-inline-block mr-1"
        >
          <bb-icon-ui
            size="md"
            backgroundType="circle"
            name="category-{{ hostRef.transformCategoryToIconName(cssClass) }}"
          ></bb-icon-ui>
        </span>
        {{ name }}
      </span>
      <div>
        <div class="row">
          <strong>
            <bb-amount-ui
              class="h5 my-1 col-12 chart-tooltip-analysis-amount"
              [amount]="totalAmount.amount"
              [currency]="totalAmount.currencyCode"
            ></bb-amount-ui>
          </strong>
        </div>
        <p
          *ngIf="totalAmount.amount !== undefined"
          class="chart-tooltip-analysis-portion text-muted"
          i18n="Slice portion|Active slice portion of total amount@@donut-chart.tooltip.portion"
        >
          {{ portion }}% of total
        </p>
        <ng-container *ngIf="trend !== undefined">
          <hr class="chart-tooltip-divider p-0">
          <p
            class="chart-tooltip-analysis-trend"
            [ngClass]="hostRef.textClass(trend, isSpending)"
          >
            <span class="d-flex align-items-center d-block mt-1">
              <bb-icon-ui
                class="mr-1"
                [name]="hostRef.iconClass(trend)"
              ></bb-icon-ui>
              <span class="h5 m-0 d-block">
                <ng-container *ngIf="hostRef.abs(trend) >= 1">
                  {{ hostRef.abs(trend) }}%
                </ng-container>
              </span>
            </span>
            <span class="chart-tooltip-info-text text-muted w-100">
              <ng-container
                *ngIf="trend <= -1"
                i18n="Less than previous month|Text to describe that trend is negative@@donut-chart.tooltip.trend.less"
              >
                less than previous month
              </ng-container>
              <ng-container
                *ngIf="trend >= 1"
                i18n="More than previous month|Text to describe that trend is positive@@donut-chart.tooltip.trend.more"
              >
                more than previous month
              </ng-container>
              <ng-container
                *ngIf="trend > -1 && trend < 1"
                i18n="Same as previous month|Text to describe that trend is around zero@@donut-chart.tooltip.trend.same"
              >
                Same as previous month
              </ng-container>
            </span>
          </p>
        </ng-container>
      </div>
    </div>
  </ng-template>
</ng-template>
