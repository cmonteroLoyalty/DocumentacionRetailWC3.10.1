Find branches and ATMs

Preferences

The widget has the following preferences that managers or administrators configure in Experience Manager:

Title and description Name Admin[a] Default value

Google maps API key input

Set the Google Maps API key to communicate with Google services. Your API key must include the following Google APIs: Maps JavaScript API, Geocoding API, and Places API. Google maps API key can be provided from app level as input by setting the value to app-config googleAPIKey.

input.googleAPIKey

X

empty

API Key (deprecated)

Set Google maps API for the widget. This property is deprecated. API key should be set through googleAPIKey input instead.

apiKey

X

empty

Default zoom value for a map

Set the initial map zoom level.

mapZoom

X

15

Default latitude

Set the default latitude that the widget uses for map center and server request if it fails to establish the customer’s location.

latitude

X

52.3760992

Default longitude

Set the default longitude that the widget uses for map center and server request if it fails to establish the customer’s location.

longitude

X

4.900779

Radius in km to filter list view

Set the radius from the current location within which the widget shows branches in the list, if the map view is removed.

placesFilterRadius

X

2

Maximum amount of locations to display

Limit the number of branches shown in the list to this number, if the map view is removed.

limitList

X

500

Measurement system

Set if the distance of the location from the user’s current location is displayed in metric or in US imperial measurement units.

measurementSystem

X

metric

[a] Preferences that have this column checked, are only editable for administrators.

Extension slots

This widget has the following extension slots:

  1. bbPlacesContainerCustomizable

  2. bbPlacesSearchCustomizable

  3. bbPlacesSearchByTypeCustomizable

  4. bbPlacesListCustomizable

  5. bbPlacesMapCustomizable

  6. bbPlacesCurrentLocationCustomizable

  7. bbPlaceHeaderCustomizable

  8. bbPlaceBodyCustomizable

  9. bbPlacesViewsButtonsGroupCustomizable

  10. bbPlacesInfoWindowCustomizable

bbPlacesContainerCustomizable

This extension slot includes the following default template:

<ng-template bbPlacesContainerCustomizable let-hostRef let-places="context">
  <div class="card">
    <div class="card-body">
      <div class="row" [class.bb-stack--center]="hostRef.error">
        <ng-container *ngIf="!hostRef.error; else errorState">
          <div class="col-md-6">
            <ng-container bbPlacesSearchCustomizable [bbHostRef]="hostRef"></ng-container>
            <ng-container bbPlacesCurrentLocationCustomizable [bbHostRef]="hostRef"></ng-container>
            <ng-container bbPlacesSearchByTypeCustomizable [bbHostRef]="hostRef" *ngIf="places && places.length"></ng-container>
            <ng-container bbPlacesViewsButtonsGroupCustomizable [bbHostRef]="hostRef"></ng-container>
            <div [hidden]="hostRef.selectedView === 'map'" class="d-md-block">
              <ng-container bbPlacesListCustomizable [bbHostRef]="hostRef" [bbTemplateContext]="places">
              </ng-container>
            </div>
          </div>
          <div class="col-md-6 d-md-block" [hidden]="hostRef.selectedView === 'list'">
            <ng-container bbPlacesMapCustomizable [bbHostRef]="hostRef" [bbTemplateContext]="places">
            </ng-container>
          </div>
        </ng-container>
      </div>
    </div>
  </div>

  <ng-template bbPlacesViewsButtonsGroupCustomizable let-hostRef>
    <div class="bb-text-align-center bb-block bb-block--md d-md-none" #groupBtns>
      <div class="btn-group btn-group-toggle bb-block bb-block--full-width">
        <button
        class="btn btn-secondary"
        [ngClass]="{'active': hostRef.selectedView === 'list'}"
        name="list"
        (click)="hostRef.selectedView = 'list'"
        i18n="List view button|List view @@places.toggle.view.list"
      >
        List
      </button>
      <button
        class="btn btn-secondary"
        [ngClass]="{'active': hostRef.selectedView === 'map'}"
        (click)="hostRef.selectedView = 'map'"
        i18n="Map view Button|Map view @@places.toggle.view.map"
      >
        Map
      </button>
      </div>
    </div>
  </ng-template>

  <ng-template bbPlacesCurrentLocationCustomizable let-hostRef>
    <div class="bb-inline-stack bb-text-support bb-block bb-block--md" *ngIf="hostRef.currentLocation">
      <bb-icon-ui name="my-location" class="bb-inline-stack__item"></bb-icon-ui>
      <span
        class="bb-inline-stack__item"
        i18n="Current Location|current location message for places@@places.state.currentLocation"
      >Current Location - {{hostRef.currentLocation}}</span>
    </div>
  </ng-template>

  <ng-template bbPlacesSearchCustomizable let-hostRef>
    <bb-map-search-ui
      [config]="hostRef.config"
      (mapSearch)="hostRef.handleMapSearch($event)"
    ></bb-map-search-ui>
  </ng-template>

  <ng-template bbPlacesSearchByTypeCustomizable let-hostRef>
    <form
      #placeTypeFilterForm="ngForm"
      class="bb-block bb-block--md bb-stack bb-stack--density-xl"
    >
      <bb-input-checkbox-ui
        *ngFor="let type of hostRef.placeTypes; let i = index"
        ngModel
        [name]="type"
        class="bb-stack__item"
        (change)="hostRef.handleTypeSearch(hostRef.filterTypes(placeTypeFilterForm.value))"
      >
        <span i18n="filters|filters for types@@places.filter.types">
          {type, select, atm {ATM's} branch {Branches}}
        </span>
      </bb-input-checkbox-ui>
    </form>
  </ng-template>

  <ng-template bbPlacesListCustomizable let-hostRef let-placesList="context">
    <div
      class="bb-places-list"
      [ngClass]="placesList && placesList.length ? '': 'bb-places-list--empty'"
    >
      <ng-container *ngIf="placesList; else loadingState">
        <ng-container *ngIf="placesList.length; else emptyState">
          <bb-places-list
            [radius]="hostRef.radius"
            [locations]="placesList"
            [selectedPlaceId]="hostRef.selectedPlaceId | async"
            [currentPosition]="hostRef.currentPosition"
            [isMetric]="hostRef.isMetric"
            (select)="hostRef.updatePlaceSelection($event)"
          ></bb-places-list>
        </ng-container>
      </ng-container>
    </div>
  </ng-template>

  <ng-template bbPlacesMapCustomizable let-hostRef let-markers="context">
    <bb-map-view
      [apiKey]="hostRef.apiKey"
      [config]="hostRef.config"
      [mapZoom]="hostRef.mapZoom"
      [latitude]="hostRef.latitude"
      [longitude]="hostRef.longitude"
      [icons]="hostRef.icons | async"
      [locations]="markers"
      [enableInfoWindow]="hostRef.smallScreen"
      [selectedPlaceId]="hostRef.selectedPlaceId | async"
      (update)="hostRef.locationUpdate($event)"
      (select)="hostRef.updatePlaceSelection($event)"
      (mapReady)="hostRef.onMapReady()"
    ></bb-map-view>
  </ng-template>
  <!-- Empty List -->
  <ng-template #emptyState>
    <bb-empty-state-ui
      class="bb-text-align-center"
      i18n-title="No locations found|Title of a message that is shown when there are no places@@places.state.noResultsFound.title"
      i18n-subtitle="We could not find any items in this area. Try moving the map or searching for a city or branch|Subtitle of a message that is shown when there are no places@@places.state.noResultsFound.subtitle"
      title="No Locations Found"
      subtitle="We couldn’t find any branches or ATMs in this region. Try moving the map or searching for a city or branch."
      iconSize="xxl"
      iconModifier="place"
    ></bb-empty-state-ui>
  </ng-template>

  <!-- Loading State -->
  <ng-template #loadingState>
    <bb-loading-indicator-ui
      i18n-text="Loading places data|Text to display while loading places@@places.state.loadingPlaces"
    ></bb-loading-indicator-ui>
  </ng-template>

  <!-- Error State -->
  <ng-template #errorState>
    <bb-common-error-state-ui
      [error]="errorCustomTemplate"
    >
      <ng-template #errorCustomTemplate>
        <bb-empty-state-ui
          class="bb-text-align-center"
          i18n-title="Page couldn’t load|Title of a message that is shown when page couldn't load@@places.state.pageCouldnotLoad.title"
          i18n-subtitle="Try to reload the page or contact the system administrator.|Subtitle of a message that is shown when page could not load@@places.state.pageCouldnotLoad.subtitle"
          title="Page couldn’t load"
          subtitle="Try to reload the page or contact the system administrator."
          iconSize="xxl"
          iconModifier="error"
        ></bb-empty-state-ui>
      </ng-template>
    </bb-common-error-state-ui>
  </ng-template>
</ng-template>

  <div class="bb-stack bb-place-info">
    <div class="bb-inline-stack bb-inline-stack--density-sm">
      <bb-icon-ui
        [name]="hostRef.getPlaceTypeIcon(place.placeType)"
        size="md"
        class="bb-inline-stack__item"
      ></bb-icon-ui>
      <span class="bb-text-semi-bold">{{place.name}}</span>
    </div>

    <div *ngIf="place.address">
      <span>{{place.address.addressLine1}}</span>
      <span *ngIf="place.address.addressLine2">,
        <span>{{place.address.addressLine2}}</span>
      </span>
    </div>

    <div class="bb-text-support bb-inline-stack" *ngIf="place.distance !== undefined">
