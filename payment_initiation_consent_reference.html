Payment initiation consent

This page includes the following information:

  • Selector name

  • Required imports

  • Extension slots

  • API

  • Implementation example

  • Preferences

Selector name

This widget has the following selector name: bb-payment-request-widget

Required imports

The following imports are required to set up and customize this widget:

  • Widget module import

    Import the following modules into the module of your wrapper widget:

    import { PaymentRequestWidgetModule } from '@backbase/payment-request-widget-ang';

  • Data module imports

    Import the following data modules into the module of your wrapper widget:

    import { ConsentDataHttpModule } from '@backbase/data-ang/consent';

Extension slots

This widget has the following extension slots:

  1. bbPaymentRequestWidgetCustomizable

  2. bbPaymentRequestTppInformationCustomizable

  3. bbPaymentRequestAmountCustomizable

  4. bbPaymentRequestAccountSelectorCustomizable

bbPaymentRequestWidgetCustomizable

You configure this extension slot using:

<ng-container
  bbPaymentRequestWidgetCustomizable
  [bbHostRef]="hostRef"
  [bbTemplateContext]="paymentRequest | async">
</ng-container>

<ng-template
  bbPaymentRequestWidgetCustomizable
  let-hostRef
  let-paymentRequest="context">
  <ng-container *ngIf="hostRef.paymentRequestId | async; else errorTemplate">
    <ng-container *ngIf="!(hostRef.loading | async) as loading; else loadingTemplate">
      <ng-container *ngIf="!(hostRef.error | async); else errorTemplate">
        <div class="bb-payment-request" *ngIf="paymentRequest">
          <div class="bb-heading-2 bb-block bb-block--md" i18n="Widget header|Payment request access message@@payment-request.label.header"
              data-role="payment-request-label">
            Payment request
          </div>

          <div i18n="Widget sub header|Payment request access message@@payment-request.label.sub-header"
            class="bb-subtitle bb-block bb-block--md"
            data-role="payment-request-description">
            Third party requested a payment from you. Select an account to pay from and review payment details to proceed with payment.
          </div>

          <ng-container *ngIf="!(hostRef.saving | async); else savingLoadingTemplate">
            <bb-alert-ui
              *ngIf="hostRef.savingError | async"
              title="Unknown error"
              i18n-title="Unknown Error|Unknown Error Label@@payment-request.label.error.unknown">
              <span data-role="allow-payment-request-error"
                i18n="Try again|Try again error message@@payment-request.label.error.tryAgain">
                Try to make action again or contact your system administrator.
              </span>
            </bb-alert-ui>

            <div class="bb-block bb-block--lg">
              <bb-payment-request-tpp-information
                [tpp]="paymentRequest.tpp"
              ></bb-payment-request-tpp-information>
            </div>

            <div class="bb-block bb-block--lg">
              <bb-payment-request-amount
                [amount]="paymentRequest.amount"
                [currency]="paymentRequest.currency"
              ></bb-payment-request-amount>
            </div>

            <div class="bb-block bb-block--lg">
              <bb-payment-request-account-selector
                [showError]="hostRef.showAccountError"
                [accounts]="paymentRequest.accounts"
                [(selectedAccount)]="hostRef.selectedAccount"
              ></bb-payment-request-account-selector>
            </div>

            <div class="bb-block bb-block--lg">
              <div i18n="Payment details|Payment details message@@payment-request.label.payment-details"
                class="bb-block--sm bb-text-bold"
                data-role="payment-details-label">
                Payment details
              </div>

              <div>{{paymentRequest.details}}</div>
            </div>
          </ng-container>

          <div class="bb-button-bar bb-button-bar--reverse">
            <button
              bbButton
              [disabled]="hostRef.saving | async"
              (click)="hostRef.allowPaymentRequest()"
              data-role="confirm-button"
              i18n="Confirm button|Confirm consent request@@payment-request.label.confirm"
              color="primary"
              class="bb-button-bar__button">
              Confirm
            </button>
            <button
              bbButton
              [disabled]="hostRef.saving | async"
              (click)="hostRef.rejectPaymentRequest()"
              data-role="cancel-button"
              i18n="Cancel button|Cancel consent request@@payment-request.label.cancel"
              color="secondary"
              class="bb-button-bar__button">
              Cancel
            </button>
          </div>

          <ng-template #savingLoadingTemplate>
            <div class="bb-state-container">
              <bb-loading-indicator-ui
                text="Processing the request..."
                i18n-text="Processing|Processing the allow/reject payment request@@payment-request.label.processing"
              ></bb-loading-indicator-ui>
            </div>
          </ng-template>
        </div>
      </ng-container>
    </ng-container>

    <ng-template #loadingTemplate>
      <div class="bb-state-container">
        <bb-loading-indicator-ui></bb-loading-indicator-ui>
      </div>
    </ng-template>
  </ng-container>

  <ng-template #errorTemplate>
    <div class="bb-state-container">
      <bb-payment-request-error
        [paymentRequestId]="hostRef.paymentRequestId | async"
        [errorKind]="(hostRef.error | async)?.kind"
      ></bb-payment-request-error>
    </div>
  </ng-template>
</ng-template>

<ng-template #notificationError>
  <span i18n="Operation Failed|Error notification when failing to save payment request@@payment-request.label.notification.error">
    Operation failed
  </span>
</ng-template>

bbPaymentRequestAccountSelectorCustomizable

You configure this extension slot using:

<ng-container
  bbPaymentRequestAccountSelectorCustomizable
  [bbHostRef]="hostRef"
  [bbTemplateContext]="accounts">
</ng-container>

<ng-template
  bbPaymentRequestAccountSelectorCustomizable
  let-hostRef
  let-accounts="context">
  <div class="bb-block bb-block--sm bb-text-bold"
       data-role="from-account-label"
       i18n="From account|From account label@@payment-request.label.fromAccount">
    From account
  </div>
  <div [ngClass]="{ 'border border-danger rounded': hostRef.showError }">
    <bb-account-selector-ui
      data-role="account-selector"
      [items]="accounts"
      [disableScrollEnd]="false"
      [multiple]="false"
      [closeOnSelect]="true"
      [filterItems]="false"
      [selectedItems]="hostRef.selectedAccount"
      [dropdownPosition]="'bottom'"
      (change)="hostRef.onSelectAccount($event)"
      placeholder="Select Accounts"
      i18n-placeholder="Select Accounts placeholder@@payment-request.account-selector.placeholder"
    >
      <ng-template bbCustomSingleSelectedItemLabel let-item="item">
        <bb-product-item-basic-account-ui
          [active]="true"
          [amount]="item.balance"
          [currency]="item.currency"
          [productNumber]="item.IBAN || item.BBAN || item.number"
          [title]="item.name"
        >
          <ng-template bbCustomProductItemBasicAccountDetailsFormatter let-amount="amount" let-currency="currency">
            <bb-amount-ui
              class="font-weight-bold"
              [amount]="amount"
              [currency]="currency"
              [mapCurrency]="false"
            ></bb-amount-ui>
          </ng-template>
        </bb-product-item-basic-account-ui>
      </ng-template>
    </bb-account-selector-ui>

  </div>
  <div
    *ngIf="hostRef.showError"
    class="bb-payment-request-product-selector__no-account"
    data-role="no-accounts-error-message"
    i18n="Select an account|No account selected label@@payment-request.label.noAccountSelected">
    Select an account.
  </div>
</ng-template>

Implementation example

The following examples show the implementation of this widget:

  • Wrapper widget module

    import { NgModule } from '@angular/core';
import { CommonModule } from '@angular/common';
import { BackbaseCoreModule } from '@backbase/foundation-ang/core';
import { PaymentRequestWidgetModule } from '@backbase/payment-request-widget-ang';
import { CustomWidgetComponent } from './custom-widget.component';

@NgModule({
  declarations: [CustomWidgetComponent],
  imports: [
    CommonModule,
    BackbaseCoreModule.withConfig({
      classMap: { CustomWidgetComponent },
    }),
    PaymentRequestWidgetModule,
  ],
  exports: [
    PaymentRequestWidgetModule,
    CustomWidgetComponent,
  ],
})
export class CustomWidgetModule { }

  • Wrapper widget component

    import { Component, EventEmitter, Output } from '@angular/core';

@Component({
  selector: 'bb-custom-widget',
  templateUrl: './custom-widget.component.html',
  styles: []
})
export class CustomWidgetComponent {
  @Output() allowPaymentRequestRedirectUrl = new EventEmitter<string>();
  @Output() rejectPaymentRequestRedirectUrl = new EventEmitter<string>();
}

  • Custom template

    <bb-payment-request-widget
  (allowPaymentRequestRedirectUrl)="allowPaymentRequestRedirectUrl.emit($event)"
  (allowPaymentRequestRedirectUrl)="rejectPaymentRequestRedirectUrl.emit($event)"
></bb-payment-request-widget>
