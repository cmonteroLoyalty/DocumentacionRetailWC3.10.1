Account information dashboard and revoke

This page includes the following information:

  • Selector name

  • Required imports

  • Extension slots

  • API

  • Implementation example

  • Preferences

Selector name

This widget has the following selector name: consent-details-widget-ang

Required imports

The following imports are required to set up and customize this widget:

  • Widget module import

    Import the following modules into the module of your wrapper widget:

    import { ConsentDetailsWidgetModule } from '@backbase/consent-details-widget-ang';

  • Data module imports

    Import the following data modules into the module of your wrapper widget:

    import { ConsentDataHttpModule } from '@backbase/data-ang/consent';

Extension slots

This widget has the following extension slots:

  1. bbConsentDetailsContainerCustomizable

  2. bbConsentDetailsHeaderCustomizable

  3. bbConsentDetailsTppDetailsCustomizable

  4. bbConsentDetailsAccountsListCustomizable

  5. bbConsentDetailsPermissionsListCustomizable

bbConsentDetailsContainerCustomizable

You configure this extension slot using:

<ng-container
  bbConsentDetailsContainerCustomizable
  [bbHostRef]="hostRef"
  [bbTemplateContext]="{
    consentDetails: consentDetails | async,
    loading: loading | async,
    error: error | async
  }">
</ng-container>

<ng-template
  bbConsentDetailsContainerCustomizable
  let-hostRef
  let-consentDetails="context.consentDetails"
  let-error="context.error"
  let-loading="context.loading">
  <ng-container *ngIf="hostRef.consentId | async; else errorTemplate">
    <ng-container *ngIf="!loading; else loadingTemplate">
      <ng-container *ngIf="!error; else errorTemplate">
        <bb-consent-details-header
          [expiryDate]="consentDetails?.expiryDate"
          [status]="consentDetails?.status"
          [tpp]="consentDetails?.tpp"
          (revokeConsent)="hostRef.openRejectConsentModal()">
        </bb-consent-details-header>

        <div class="bb-consent-details">
          <!-- Large Viewports -->
          <div class="bb-consent-details__lg">
            <bb-consent-third-party-details
              [tpp]="consentDetails?.tpp">
            </bb-consent-third-party-details>
            <div class="bb-consent-details__lg-wrapper">
              <div class="bb-consent-details__lg-wrapper-item">
                <bb-consent-details-accounts-list
                  [accounts]="consentDetails?.accounts">
                </bb-consent-details-accounts-list>
              </div>
              <bb-consent-details-permissions-list
                [permissions]="consentDetails?.permissions">
              </bb-consent-details-permissions-list>
            </div>
          </div>

          <!-- Mobile Viewports -->
          <div class="bb-consent-details__sm">
            <div class="bb-block bb-block--lg">
              <div class="bb-block bb-block--md">
                <div
                  i18n="Expired on|Consent is expired@@consent-details.container.heading.expiredOn"
                  class="bb-text-bold" *ngIf="consentDetails?.status === hostRef.Status.EXPIRED">
                  Expired on
                </div>
                <div
                  i18n="Expiry Date|Expiry Date of Consent@@consent-details.container.heading.expiryDate"
                  class="bb-text-bold" *ngIf="consentDetails?.status !== hostRef.Status.EXPIRED">
                  Expiry date
                </div>
                <div>
                  {{consentDetails.expiryDate | bbDate:'bbShort'}}
                </div>
              </div>
              <div class="bb-block bb-block--md">
                <bb-consent-details-accounts-list
                  [accounts]="consentDetails?.accounts">
                </bb-consent-details-accounts-list>
              </div>
              <bb-consent-details-permissions-list
                [permissions]="consentDetails?.permissions">
              </bb-consent-details-permissions-list>
            </div>
            <bb-collapsible-ui>
              <ng-template
                bbCollapsibleHeader
                let-toggle="toggle">
                <div class="bb-block bb-block--md">
                  <div
                    (click)="toggle()"
                    class="bb-consent-details__sm-collapsible">
                    <span
                      class="bb-text-bold"
                      i18n="See other details|Third party details@@consent-details.container.label.seeDetails">
                      See other details
                    </span>
                    <bb-icon-ui name="toggle-down" size="sm">
                    </bb-icon-ui>
                  </div>
                </div>
              </ng-template>
              <ng-template bbCollapsibleBody>
                <bb-consent-third-party-details
                  [tpp]="consentDetails?.tpp">
                </bb-consent-third-party-details>
              </ng-template>
            </bb-collapsible-ui>

            <!-- Small Viewport Revoke Button -->
            <div
              *ngIf="consentDetails?.status === hostRef.Status.ACTIVE"
              class="bb-consent-details__button bb-button-bar">
              <button
                bbButton
                class="bb-button-bar__button"
                (click)="hostRef.openRejectConsentModal()"
                color="danger"
                i18n="Revoke Consent|Revoke consent button@@consent-details.header.button.revoke-consent">
                Revoke Consent
              </button>
            </div>
          </div>
        </div>

        <bb-modal-ui
          [isOpen]="hostRef.isModalOpen">
          <bb-modal-header-ui>
            <ng-template bbCustomModalHeader>
              <span
                class="bb-text-bold"
                data-role="dialog-title"
                i18n="Revoke Consent?|Revoke consent heading@@consent-details.container.heading.revoke-consent"
                bbConfirmTitle>
               Revoke Consent?
              </span>
            </ng-template>
          </bb-modal-header-ui>
          <bb-modal-body-ui>
            <ng-template bbCustomModalBody>
              <div
                data-role="dialog-question"
                i18n="Are you sure you want to permanently revoke consent?
                This action can't be undone.|
                Revoke Consent body@@consent-details.container.body.revoke-consent"
                bbConfirmBody>
                Are you sure you want to permanently revoke consent for {{ consentDetails?.tpp.name }}?
                This action can't be undone.
              </div>
            </ng-template>
          </bb-modal-body-ui>
          <bb-modal-footer-ui>
            <ng-template bbCustomModalFooter>
              <button
                bbButton
                data-role="cancel-button"
                color="secondary"
                (click)="hostRef.dismissRejectConsentModal()"
                i18n="Cancel|Cancel and close modal@@consent-details.container.button.label.cancel">
                Cancel
              </button>
              <button
                bbButton
                color="danger"
                data-role="revoke-button"
                (click)="hostRef.revokeConsent()"
                i18n="Revoke|Approve revoking of consent@@consent-details.container.button.label.revoke">
                Revoke
              </button>
            </ng-template>
          </bb-modal-footer-ui>
        </bb-modal-ui>
      </ng-container>
    </ng-container>

    <ng-template #loadingTemplate>
      <div class="bb-state-container">
        <bb-loading-indicator-ui loadersize="lg"></bb-loading-indicator-ui>
      </div>
    </ng-template>
  </ng-container>
  <ng-template #errorTemplate>
    <div class="bb-state-container">
      <bb-consent-details-error
        [consentId]="hostRef.consentId | async"
        [errorKind]="error?.kind">
      </bb-consent-details-error>
    </div>
  </ng-template>
</ng-template>

<ng-template #notificationError>
  <span i18n="Failed to revoke consent|Error notification message@@consent-details.container.notification.error">
    Failed to revoke consent
  </span>
</ng-template>

<ng-template #notificationSuccess>
  <span i18n="Successfully revoked consent
              |Success notification message@@consent-details.container.notification.success">
    Consent was revoked
  </span>
</ng-template>

Object property Description

Component context.

Consent details.

Loading spinner visibility flag.

readonly error: ConsentDetailsError | undefined

Consent details request errors.

bbConsentDetailsHeaderCustomizable

You configure this extension slot using:

<ng-container
  bbConsentDetailsHeaderCustomizable
  [bbHostRef]="hostRef"
  [bbTemplateContext]="{
    tpp: tpp,
    status: status,
    expiryDate: expiryDate
  }">
</ng-container>

<ng-template
  bbConsentDetailsHeaderCustomizable
  let-hostRef
  let-tpp="context.tpp"
  let-status="context.status"
  let-expiryDate="context.expiryDate">
  <div class="bb-consent-details-header bb-stack">
    <div class="bb-stack__item">
      <bb-third-party-provider-ui
        [name]="tpp?.name"
        [url]="tpp?.website"
        [image]="tpp?.imageUrl"
        [verified]="tpp?.isTrusted"
        [condensed]=true>
      </bb-third-party-provider-ui>
    </div>

    <!-- Large Viewports -->
    <div
      data-role="status-badge-large"
      class="bb-stack__item">
      <bb-badge-ui
        data-role="status-badge-active"
        i18n-text="Active|Consent status active@@consent-details.header.badge.active"
        *ngIf="status === hostRef.Status.ACTIVE"
        color="success"
        text="Active">
      </bb-badge-ui>
      <bb-badge-ui
        data-role="status-badge-expired"
        i18n-text="Expired|Consent status expired@@consent-details.header.badge.expired"
        *ngIf="status === hostRef.Status.EXPIRED"
        color="secondary"
        text="Expired">
      </bb-badge-ui>
      <bb-badge-ui
        data-role="status-badge-revoked"
        i18n-text="Revoked|Consent status revoked@@consent-details.header.badge.revoked"
        *ngIf="status === hostRef.Status.REVOKED"
        color="danger"
        text="Revoked">
      </bb-badge-ui>
    </div>

    <div class="bb-consent-details-header__status-wrapper bb-stack__item bb-stack__item--push-right">
      <div class="bb-consent-details-header__status bb-text-align-right">
        <div
          i18n="Expired on|Consent is expired@@consent-details.header.heading.expiredOn"
          class="bb-text-bold" *ngIf="status === hostRef.Status.EXPIRED">Expired on</div>
        <div
          data-role="token-expiry-label-active"
          i18n="Expiry Date|Expiry Date of Consent@@consent-details.header.heading.expiryDate"
          class="bb-text-bold" *ngIf="status !== hostRef.Status.EXPIRED">Expiry date</div>
        <div data-role="expiry-date">
          {{expiryDate | bbDate:'bbShort'}}
        </div>
      </div>
      <button
        bbButton
        *ngIf="status === hostRef.Status.ACTIVE"
        (click)="hostRef.onRevokeConsent()"
        color="danger"
        data-role="revoke-consent-button"
        type="submit">
        <span i18n="Revoke Consent|Revoke consent button@@consent-details.header.button.revoke-consent">
          Revoke Consent
        </span>
      </button>
    </div>
  </div>
</ng-template>
