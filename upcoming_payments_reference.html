Manage upcoming payments

Preferences

The widget has the following preferences that managers or administrators can edit in Experience Manager:

Title and description Name Admin[a] Default value Data type

Number of payments per page

pageSize

50

long

Number of seconds to dismiss the notification

dismissMessageTime

X

3

long

[a] Preferences that have this column checked, are only editable for administrators.

Extension slots

This widget has the following extension slots:

  1. bbUpcomingPaymentsContainerCustomizableDirective

  2. bbUpcomingPaymentsListCustomizableDirective

  3. bbUpcomingPaymentsItemGroupCustomizableDirective

  4. bbUpcomingPaymentsItemCustomizableDirective

  5. bbUpcomingPaymentsListLoadMoreCustomizableDirective

  6. bbUpcomingPaymentsDetailsCustomizableDirective

  7. bbUpcomingPaymentsDetailsListCustomizableDirective

  8. bbUpcomingPaymentsDetailsModalFooterCustomizableDirective

  9. bbUpcomingPaymentsCancelPaymentConfirmationCustomizableDirective

bbUpcomingPaymentsContainerCustomizableDirective

You configure this extension slot using:

<ng-template bbUpcomingPaymentsContainerCustomizableDirective let-hostRef>
  <bb-upcoming-payments-list
    *ngIf="(hostRef.upcomingPayments$ | async) as upcomingPayments; else loadingUpcomingPayments"
    [upcomingPayments]="upcomingPayments.items"
    [totalCount]="upcomingPayments.totalCount"
    [loadMoreSize]="hostRef.loadMoreSize"
    [loadMoreFrom]="hostRef.loadMoreFrom"
    [isLoadMore]="false"
    (paymentItemClick)="hostRef.onPaymentItemClick($event)"
    (loadMoreParam)="hostRef.onLoadMoreParam($event, upcomingPayments.items.length,upcomingPayments.totalCount)"
  ></bb-upcoming-payments-list>

  <ng-template #loadingUpcomingPayments>
    <bb-loading-indicator-ui text="Loading data..."></bb-loading-indicator-ui>
  </ng-template>

  <bb-upcoming-payments-details
    *ngIf="hostRef.selectedPaymentItem"
    [isOpen]="hostRef.openDetails"
    (close)="hostRef.onCloseDetails()"
    [payment]="hostRef.selectedPaymentItem"
    (cancelPayment)="hostRef.onCancelPayment()"
    (editPayment)="hostRef.onEditPayment()"
  ></bb-upcoming-payments-details>

  <bb-upcoming-payments-cancel-payment-confirmation
    *ngIf="hostRef.openConfirmation"
    [isOpen]="hostRef.openConfirmation"
    [isRecurring]="hostRef.isRecurringPayment(hostRef.selectedPaymentItem)"
    (cancelPayment)="onConfirmCancelPayment(true)"
    (keepPayment)="onConfirmCancelPayment(false)"
  ></bb-upcoming-payments-cancel-payment-confirmation>

  <ng-template #successNotificationForRecurring>
    <span i18n="Success notification for cancelled recurring payment action|Notification message that will be shown if payment cancellation was successful@@upcoming-payments.cancel.payment.notification.recurringSuccess">
      The upcoming payment and all following payments to "{{hostRef.creditorName}}" has been cancelled successfully
    </span>
  </ng-template>

  <ng-template #successNotification>
    <span i18n="Success notification for cancelled payment action|Notification message that will be shown if payment cancellation was successful@@upcoming-payments.cancel.payment.notification.success">
      The payment to "{{hostRef.creditorName}}" has been cancelled successfully.
    </span>
  </ng-template>

  <ng-template #failureNotification>
    <span i18n="Failure notification for cancelled payment action|Notification message that will be shown if payment cancellation failed@@upcoming-payments.cancel.payment.notification.failure">
      Failed to cancel The payment to "{{hostRef.creditorName}}".
    </span>
  </ng-template>
</ng-template>

bbUpcomingPaymentsListCustomizableDirective

You configure this extension slot using:

<ng-template bbUpcomingPaymentsListCustomizableDirective let-hostRef>
  <ng-container *ngIf="hostRef.upcomingPayments && hostRef.upcomingPayments?.length > 0; else emptyState">
    <bb-upcoming-payments-item-group
      *ngFor="let paymentsGroup of hostRef.upcomingPayments; trackBy: trackPaymentsGroup"
      (paymentItemClick)="hostRef.onPaymentItemClick($event)"
      [paymentsGroup]="paymentsGroup"
    ></bb-upcoming-payments-item-group>

    <div class="d-flex justify-content-center p-2">
      <bb-load-button-ui
        color="secondary"
        i18n="Load more payment|Load more button for loading more payments@@upcoming-payments.load.more"
        [isLoading]="hostRef.isLoadMore"
        [size]="hostRef.loadMoreSize"
        [from]="hostRef.loadMoreFrom"
        [totalCount]="hostRef.totalCount"
        (paramChange)="hostRef.loadMoreParamChange($event)"
      >
        Load more
      </bb-load-button-ui>
    </div>
  </ng-container>

  <ng-template #emptyState>
    <bb-empty-state-ui
      title="No Upcoming Payments"
      i18n-title="@@upcoming-payments.no.payments.title"
      subtitle="You do not have any upcoming payments to be displayed"
      i18n-subtitle="@@upcoming-payments.no.payments.subTitle"
      iconSize="xl"
      iconModifier="swap-horiz"
    ></bb-empty-state-ui>
  </ng-template>
</ng-template>

bbUpcomingPaymentsItemGroupCustomizableDirective

You configure this extension slot using:

<ng-template bbUpcomingPaymentsItemGroupCustomizableDirective let-hostRef>
  <!-- Execution date -->
  <div
    tabindex="0"
    class="bb-subheader bb-text-support bb-block bb-block--lg"
    i18n="@@upcoming-payments.group.header"
    data-role="group-item-label"
  >
    {hostRef.formatRequestedExecutionDate(), select, today {Today} tomorrow {Tomorrow} other
      {{{ hostRef.paymentsGroup?.requestedExecutionDate | date:'longDate'}}} }
  </div>
  <!-- group of payments -->
  <ng-container *ngIf="hostRef.paymentsGroup?.items?.length > 0">
    <div class="bb-block bb-block--lg">
      <ul class="bb-list">
        <li
          *ngFor="let payment of hostRef.paymentsGroup?.items; trackBy: hostRef.trackPaymentItem"
          class="bb-list__item"
        >
          <div
            data-role="group-item-row"
            role="button"
            tabindex="0"
            (keydown.enter)="hostRef.onPaymentItemClick(payment)"
            (click)="hostRef.onPaymentItemClick(payment)"
          >
            <bb-upcoming-payments-item
              [payment]="payment">
            </bb-upcoming-payments-item>
          </div>
        </li>
      </ul>
    </div>
  </ng-container>
</ng-template>

<ng-template bbUpcomingPaymentsItemCustomizableDirective let-hostRef>

        </span>

          {{
            hostRef.transactionInformation?.counterparty?.role === 'DEBTOR'
              ? hostRef.payment?.originator.name
              : hostRef.transactionInformation?.counterparty?.name
          }}
        </div>

        <ng-container
          *ngIf="hostRef.payment && hostRef.payment.paymentMode && hostRef.payment.paymentMode === 'RECURRING'"
        >
          <span class="sr-only" i18n="screenReader@@upcoming-payments.group.item.paymentMode">
            ; Payment Mode:
          </span>

              {{ hostRef.payment.paymentMode + ' ' + hostRef.schedule | titlecase }}
            </span>
          </div>
        </ng-container>
      </div>
      <div class="bb-stack__item bb-text-bold">
        <span class="sr-only" i18n="screenReader@@upcoming-payments.group.item.amount">; Amount: </span>
        <bb-amount-ui
          data-role="upcoming-payment-item-amount"
