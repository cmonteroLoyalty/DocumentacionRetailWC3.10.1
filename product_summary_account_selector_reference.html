Selector name

This widget has the following selector name: <bb-product-summary-account-selector-widget></bb-product-summary-account-selector-widget>

Required imports

The following imports are required to set up and customize this widget:

  • Widget module import

    Import the following modules into the module of your wrapper widget:

    import { ProductSummaryAccountSelectorWidgetModule } from '@backbase/product-summary-account-selector-widget-ang';

Extension slots

This widget has the following extension slots:

If multiselect is not enabled for this widget in Experience Manager.

If multiselect is enabled for this widget in Experience Manager.

  1. bbAccountSelectorDropdownCustomizable

  2. bbAccountSelectorItemCustomizable

  3. bbAccountSelectorMultiSelectedItemsCustomizable

  4. bbAccountSelectorLoadingCustomizable

  5. bbAccountSelectorStateCustomizable

bbAccountSelectorDropdownCustomizable

This extension slot includes the following default template:

<ng-template
  bbAccountSelectorDropdownCustomizable
  let-hostRef
  let-accounts="context"
>
  <bb-account-selector-ui
    [items]="accounts"
    [dropdownPosition]="'bottom'"
    [selectedItems]="hostRef.accountSelected"
    [multiple]="hostRef.multiple | async"
    [closeOnSelect]="(hostRef.multiple | async) === false"
    (change)="hostRef.onAccountSelected($event)"
    [filterItems]="hostRef.filterItems | async"
    [compareItemsWith]="hostRef.compareProductsById.bind(hostRef)"
    [searchFunc]="hostRef.isInProductAlias.bind(hostRef)"
    data-role="accountSelector"
    placeholder="Select Account(s)"
    i18n-placeholder="Select Account(s)|Select Accounts placeholder
      @@product-summary-account-selector.empty.state.placeholder"
  >
    <ng-template
      bbCustomSingleSelectedItemLabel
      let-item="item"
    >
      <ng-container
        bbAccountSelectorItemCustomizable
        [bbHostRef]="hostRef"
        [bbTemplateContext]="item"
      >
      </ng-container>
    </ng-template>

    <ng-template
      bbCustomMultiSelectedItemsLabel
      let-items="items">
      <ng-container
        bbAccountSelectorMultiSelectedItemsCustomizable
        [bbHostRef]="hostRef"
        [bbTemplateContext]="{items: items, accounts: accounts}">
      </ng-container>
    </ng-template>

    <ng-template
      bbCustomOptionItem
      let-item="item"
      let-item$="item$"
      let-index="index"
    >
      <ng-container *ngIf="hostRef.multiple | async">
        <div class="bb-stack">
          <div class="bb-stack__item multiple-accounts-item-checkbox">
            <bb-input-checkbox-ui
              [id]="index"
              [ngModel]="item$.selected"
              (click)="$event.preventDefault()"
            >
            </bb-input-checkbox-ui>
          </div>
          <div class="bb-stack__item bb-stack__item--fill bb-ellipsis bb-ellipsis--single-line">
            <ng-container
              bbAccountSelectorItemCustomizable
              [bbHostRef]="hostRef"
              [bbTemplateContext]="item"
            >
            </ng-container>
          </div>
        </div>
      </ng-container>
      <ng-container *ngIf="(hostRef.multiple | async) === false">
        <ng-container
          bbAccountSelectorItemCustomizable
          [bbHostRef]="hostRef"
          [bbTemplateContext]="item"
        >
        </ng-container>
      </ng-container>
    </ng-template>
  </bb-account-selector-ui>
</ng-template>

This extension slot consumes the following object properties from its custom context:

Object property Description

context: Product[]

The list of accounts to display in the dropdown.

bbAccountSelectorMultiSelectedItemsCustomizable

This extension slot includes the following default template:

<ng-template
  bbAccountSelectorMultiSelectedItemsCustomizable
  let-hostRef
  let-items="context.items"
  let-accounts="context.accounts"
>
  <div class="bb-account-selector__selected">
    <div class="bb-account-selector__selected-accounts-number bb-text-bold">
      <span i18n="Selected Accounts|Selected Accounts multi label @@bb-product-summary-account-selector.multi.label">
        { items.length, plural, =1 {1 account selected} other {{{accounts.length === items.length ? 'All' : items.length}} accounts selected}}
      </span>
    </div>
    <div class="bb-account-selector__selected-accounts bb-text-subtitle">
      <ng-container *ngIf="accounts.length === items.length; else selectedAccountsTemplate">
        {{ items.length }}
        <span
          i18n="
             All Selected Accounts|All Selected Accounts label @@product-summary-account-selector.allSelectedAccounts.label
          "
          class="bb-account-selector__selected-account"
        >
          accounts
        </span>
      </ng-container>
      <ng-template #selectedAccountsTemplate>
        <bb-ellipsis-ui>
          <ng-container *ngFor="let item of items; let i = index">
                <span class="bb-account-selector__selected-account"
                >{{ hostRef.getAlias(item)
                  }}<span *ngIf="i !== items.length - 1" class="bb-account-selector__selected-account-separator">, </span>
                </span>
          </ng-container>
        </bb-ellipsis-ui>
      </ng-template>
    </div>
  </div>
</ng-template>

This extension slot consumes the following object properties from its custom context:

Object property Description

accounts: Product[]

The list of accounts to display in the dropdown.

items: Product[]

List of selected accounts.

This extension slot includes the following default template:

<ng-template
  bbAccountSelectorLoadingCustomizable
  let-hostRef
>
  <ng-container *ngIf="hostRef.error | async as error; else loading">
    <bb-empty-state-ui
      title=""
      [subtitle]="hostRef.processErrorMessages(error.messages)"

  </ng-container>
  <ng-template #loading>

      [text]="'Loading Accounts...'"
    ></bb-loading-indicator-ui>
  </ng-template>
</ng-template>

bbAccountSelectorStateCustomizable

This extension slot includes the following default template:

<ng-template
  bbAccountSelectorStateCustomizable
  let-hostRef
>
  <bb-account-state-ui
    *ngIf="hostRef.selectedAccountState"
    [isDisplayingAlert]="(hostRef.showAccountState$ | async) && ((hostRef.multiple | async) === false)"
    [item]="hostRef.selectedAccountState"
  >
  <span i18n="@@product-summary-account-selector.account-state.message">
      {hostRef.selectedAccountState?.severity, select, warning {This account is temporarily
        inactive. For more info please contact your bank.} error {This account is permanently disabled. For more info please
        contact your bank.}}
  </span>
  </bb-account-state-ui>
</ng-template>

API

The following table shows the public class members of the host component that you access from each extension slot:

Component Extension slot identifier Public class member Description

ProductSummaryAccountSelectorWidgetComponent

bbAccountSelectorDropdownCustomizable bbAccountSelectorStateCustomizable bbAccountSelectorMultiSelectedItemsCustomizable bbAccountSelectorLoadingCustomizable bbAccountSelectorStateCustomizable

filterItems: BehaviorSubject<boolean>

Enables bank customers to filter the arrangements that are presented in the dropdown. Set to true to show Filter. Set to false to show Search.

showAccountState$: BehaviorSubject<boolean>

Defines whether the account status is displayed for the selected account.

onAccountSelected(item: Product | Product[]): void

This method is used to set selected accounts.

compareProductsById(product1: Product, product2: Product): boolean

This method compares products to ensure if the IDs correspond.

isInProductAlias(value: string, product: Product): boolean

Returns true if product alias includes value, otherwise false.

accountSelected: Product[] | Product | undefined

A selected account, or a list of accounts.

error: Observable<ProductKindsError | undefined>

Used to display the error occurred during loading of accounts.

processErrorMessages(messages: string[] | undefined): string | undefined

The method processed error messages to display in the template.

getAlias(account: any): string | undefined

A method to return the alias of the given account.

selectedAccountState: AccountState | undefined

The property is used to display a message about the selected account state below the account selector.

multiple: BehaviorSubject<boolean>

Sets whether the bank customer can select multiple accounts at a time.

Implementation example

The following examples show the implementation of this widget:

  • Wrapper widget module

    product-summary-account-selector-widget-extended.module.ts
    import { NgModule } from '@angular/core';
import { CommonModule } from '@angular/common';
import { FormsModule } from '@angular/forms';
import { BackbaseCoreModule } from '@backbase/foundation-ang/core';
import { ProductSummaryAccountSelectorWidgetModule } from '@backbase/product-summary-account-selector-widget-ang';
import { ProductSummaryAccountSelectorWidgetExtendedComponent } from './product-summary-account-selector-widget-extended.component';

@NgModule({
  declarations: [ProductSummaryAccountSelectorWidgetExtendedComponent],
  imports: [
    CommonModule,
    FormsModule,
    BackbaseCoreModule.withConfig({
      classMap: { ProductSummaryAccountSelectorWidgetExtendedComponent },
    }),
    ProductSummaryAccountSelectorWidgetModule,
  ],
  exports: [ProductSummaryAccountSelectorWidgetModule, ProductSummaryAccountSelectorWidgetExtendedComponent],
})
export class ProductSummaryAccountSelectorWidgetExtendedModule {}

  • Wrapper widget component

    product-summary-account-selector-widget-extended.component.ts
    import { Component, EventEmitter, Output } from '@angular/core';

@Component({
  selector: 'bb-product-summary-account-selector-widget-extended',
  templateUrl: './product-summary-account-selector-widget-extended.component.html',
})
export class ProductSummaryAccountSelectorWidgetExtendedComponent {
  @Output() selectedAccount = new EventEmitter<string>();
  @Output() selectedKind = new EventEmitter<string>();
}

  • Custom template

    product-summary-account-selector-widget-extended.component.html
    <bb-product-summary-account-selector-widget
  (selectedAccount)="selectedAccount.emit($event)"
  (selectedKind)="selectedKind.emit($event)"
></bb-product-summary-account-selector-widget>

Mask account numbers

Enable or disable backend account masking.

  • If enabled, the widget requests account masking from the backend service, and if the backend service supports this feature, the backend service sends a masked account number to the widget. DBS supports this feature, see this page for more information.

  • If disabled, the widget does not request account masking from the backend service.
