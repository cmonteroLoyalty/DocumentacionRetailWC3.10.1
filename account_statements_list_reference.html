  • Selector name

Selector name

This widget has the following selector name:

<bb-account-statement-list-widget></bb-account-statement-list-widget>

Required imports

The following imports are required to set up and customize this widget:

  • Widget module import

    Import the following modules into the module of your wrapper widget:

    import { AccountStatementListWidgetModule } from '@backbase/account-statement-list-widget-ang';

Extension slots

This widget has the following extension slots:

  1. bbAccountStatementListCustomizable

  2. bbAccountStatementTableCustomizable

  3. bbAccountStatementTableRowCustomizable

  4. bbAccountStatementDownloadFailureNotificationHeaderCustomizable

  5. bbAccountStatementFilterFormCustomizable

  6. bbAccountStatementDownloadsCustomizableDirective

  7. bbAccountStatementAccountSelectorCustomizableDirective

bbAccountStatementListCustomizable

This extension includes the following default template:

<ng-template
    bbAccountStatementListCustomizable
    let-host
    let-accounts="context.accounts"
    let-accountsError="context.accountsError"
    let-categories="context.categories"
    let-selectedAccount="context.selectedAccount"
    let-accountStatements="context.accountStatements"
  >
    <div aria-live="polite">
      <ng-container *ngIf="accounts; else loadingState">
        <ng-container *ngIf="!accountsError; else errorStateAccounts">
          <ng-container *ngIf="accounts.length; else emptyStateAccounts">

            <div
              class="bb-block bb-block--xl"
            >
              <bb-account-statement-account-selector
                [accounts]="accounts"
                (accountSelect)="host.selectAccount($event)"
              ></bb-account-statement-account-selector>
            </div>

            <div class="card">
              <ng-container
                *ngIf="accountStatements && host.accountStatements.isLoading !== host.loadingStates.LOADING; else loadingState"
              >
                <ng-container *ngIf="!host.accountStatements.error; else errorStateAccountStatements">
                  <ng-container
                    *ngIf="accountStatements.length
                           || host.accountStatements.areFiltersApplied
                           || host.accountStatements.isLoading === host.loadingStates.LOADING_FILTERED;
                           else emptyStateAccountStatements"
                  >
                    <bb-account-statement-filter-form
                      [categories]="categories"
                      [filtersApplied]="host.accountStatements.areFiltersApplied"
                      [filterParams]="host.filterParams | async"
                      (filtersChange)="host.setFilterParams($event)"
                      [showCategory]="host.showCategory | async"
                    ></bb-account-statement-filter-form>

                    <ng-container *ngIf="host.accountStatements.isLoading !== host.loadingStates.LOADING_FILTERED; else loadingState">
                      <ng-container *ngIf="accountStatements.length; else emptyStateAccountStatementsFiltered">
                        <bb-account-statement-table
                          class="bb-block card-section"
                          [accountStatements]="accountStatements"
                          [showCategory]="host.showCategory | async"
                          [showDescription]="host.showDescription | async"
                          [sortingParams]="host.sortingParams| async"
                          [loading]="host.accountStatements.isLoading === host.loadingStates.LOADING_TABLE"
                          (sortingParamsChange)="host.setSortingParams($event)"
                        ></bb-account-statement-table>

                        <div
                          class="bb-block bb-block--xl bb-text-align-center"
                          *ngIf="host.accountStatements.isLoading !== host.loadingStates.LOADING_TABLE"
                        >
                          <bb-load-button-ui
                            color="link"
                            [isLoading]="host.accountStatements.isLoading === host.loadingStates.LOADING_MORE"
                            [size]="host.pageSize | async"
                            [from]="(host.loadingParamsNew | async)?.from"
                            [totalCount]="host.accountStatements?.totalCount"
                            (paramChange)="host.setLoadingParams($event)"
                          >
                            <span
                              i18n="Load more|Load More button text@@accountStatement.list.load-more"
                            >Load more</span>
                          </bb-load-button-ui>
                        </div>
                      </ng-container>
                    </ng-container>
                  </ng-container>
                </ng-container>
              </ng-container>
            </div>
          </ng-container>
        </ng-container>
      </ng-container>
    </div>

    <ng-template #loadingState>
      <bb-loading-indicator-ui
        loaderSize="md"
        i18n-text="Loading data...|Loading title@@accountStatement.loading.title"
        text="Loading data..."
      ></bb-loading-indicator-ui>
    </ng-template>

    <ng-template #emptyStateAccounts>
      <div class="card">
        <bb-empty-state-ui
          class="bb-state-container"
          iconSize="xxl"
          iconModifier="current-accounts"
          i18n-title="No Account selected|Accounts state title theme two@@accountStatement.accounts.state.title.theme.two"
          title="No account selected"
          i18n-subtitle="You need to select account in order to view account statements|Accounts state subtitle theme two@@accountStatement.accounts.state.subtitle.theme.two"
          subtitle="You need to select account in order to view account statements"
        ></bb-empty-state-ui>
      </div>
    </ng-template>

    <ng-template #errorStateAccounts>
      <div class="card">
        <bb-empty-state-ui
          class="bb-state-container"
          data-role="accounts-error-state"
          iconSize="xxl"
          iconModifier="error"
          subtitle="Try to reload the page or contact the system administrator."
          i18n-subtitle="Account list error state subtitle@@accountStatement.account-list.error-state.subtitle"
          title="Account list is not loading"
          i18n-title="Account list is not loading error state title@@accountStatement.account-list.error-state.title"
        >
        </bb-empty-state-ui>
      </div>
    </ng-template>

    <ng-template #emptyStateAccountStatements>
      <bb-empty-state-ui
        class="bb-state-container"
        iconSize="xxl"
        iconModifier="description"
        i18n-title="No Account Statements found|Account statements state title theme two@@accountStatement.statements.state.title.theme2"
        title="No Account Statements for selected account"
        i18n-subtitle="You don't have any account statements to be displayed for this account|Account statements state subtitle theme two@@accountStatement.statements.state.subtitle.theme2"
        subtitle="You don't have any account statements to be displayed for this account"
      ></bb-empty-state-ui>
    </ng-template>

    <ng-template #emptyStateAccountStatementsFiltered>
      <bb-empty-state-ui
        class="bb-state-container"
        iconSize="xxl"
        iconModifier="search"
        i18n-title="No Results Found|Account statements filtered state title@@accountStatement.statements.filtered.state.title"
        title="No Results Found"
        i18n-subtitle="We can’t find what you’re looking for. Please try a different criteria.|Account statements filtered state subtitle@@accountStatement.statements.filtered.state.subtitle"
        subtitle="We can’t find what you’re looking for. Please try a different criteria."
      ></bb-empty-state-ui>
    </ng-template>

    <ng-template #errorStateAccountStatements>
      <bb-empty-state-ui
        class="bb-state-container"
        data-role="account-statements-error-state"
        iconSize="xxl"
        iconModifier="error"
        subtitle="Try to reload the page or contact the system administrator."
        i18n-subtitle="Account statements list error state subtitle@@accountStatement.statement-list.error-state.subtitle"
        title="Account statement list is not loading"
        i18n-title="Account statement list is not loading error state title@@accountStatement.statement-list.error-state.title"
      >
      </bb-empty-state-ui>
    </ng-template>
  </ng-template>

Relative filter date picker limit (months)

The number of months the date picker counts back from the present day and disallows selection. For example, if the value is 2 and the present date is 01/03/2020, the first selectable date is 01/01/2020.

Absolute filter date picker limit (yyyy-mm-dd)

The absolute date the date picker disallows selection. The format has to be yyyy-mm-ddd, but / and . are accepted as separators. If this preference is set, it overrides the value in Relative filter date picker limit.
